    <h1>Live Stock Ticker</h1>

    <div id="ticker">Loading...</div>

    <script>

        async function fetchStockPrice() {

            const sheetUrl = ‘https://docs.google.com/spreadsheets/d/e/2PACX-1vStGd7P7JHT0rl105X77s6j90lBYRgeA_Md0PrjTpbqI-DgMyL2a1GRVyPcBlPXn6EtxxaxR2G04UdG/pubhtml';

           

            try {

                const response = await fetch(sheetUrl);

                const data = await response.text();

                const rows = data.split("\n");

                const latestPrice = rows[1].split(",")[1];

                document.getElementById("ticker").innerText = `AAPL: $${latestPrice}`;

            } catch (error) {

                console.error("Error fetching stock data:", error);

                document.getElementById("ticker").innerText = "Error loading data";

            }

        }

        fetchStockPrice();

        setInterval(fetchStockPrice, 5000); // Update every 5 seconds

    </script>
